```
class Solution:
    def maxDistance(self, grid: List[List[int]]) -> int:
        n = len(grid)
        queue = []
   

        moves = [[-1,0],[1,0],[0,1],[0,-1]]

        # enqueue roots (land)
        queue = collections.deque([i,j] for i in range(n) for j in range(n) if grid[i][j] == 1)#for use deque((i,j) ....)
        
        # if all lands or waters
        if len(queue) == 0 or len(queue) == n**2:
            return -1
        # recording the number of layers
        distance = -1 
        
        # iterate through the grid
        while queue:
            distance += 1
            size = len(queue)
            for i in range(size):
                r,c  = queue.popleft()                             
                for move in moves:
                    r2 = r + move[0]
                    c2 = c + move[1]
                    if 0<=r2<n and 0<=c2<n and grid[r2][c2] == 0:
                        grid[r2][c2] = 2
                        queue.append([r2,c2])        
        return distance
```

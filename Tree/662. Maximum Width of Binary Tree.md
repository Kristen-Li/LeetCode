class Solution:
    def dfs(root, pos=0, level=0):
                if not root:
                    return
                dic.setdefault(level, pos)
                self.max_width = max(self.max_width, pos - dic[level] + 1)
                dfs(root.left, pos*2, level+1)
                dfs(root.right, pos*2+1, level+1)

            self.max_width = 0
            dic = {}
            dfs(root)
            return self.max_width
